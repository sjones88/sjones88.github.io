JavaScript: Nested Objects

In the 'gradebook' example to the right:

  • The forEach() method takes a function, which takes two arguments, 'student' and 'index'. For each element in the 'students' array, the gradebook object gets 'testScores' added as a key and the 'scores' array element at 'index' added as a value.

  • The "gradebook['addScore']" statement assigns a function to an object key in 'gradebook'. The function takes two parameters - 'name' and 'score' - and pushes 'score' to 'testScores' in the 'name' key.

  • The reduce() method

Ruby Classes:

var students = ["Joseph", "Susan", "William", "Elizabeth"]

var scores = [ [80, 70, 70, 100],
               [85, 80, 90, 90],
               [75, 70, 80, 75],
               [100, 90, 95, 85] ]

var gradebook = {}
students.forEach(function(student, index) {
  gradebook[student] = {testScores: scores[index]};
})
gradebook['addScore'] = function(name, score) {
  gradebook[name].testScores.push(score);
}
gradebook['getAverage'] = function(name) {
  return average(gradebook[name].testScores);
}
var average = function(scores) {
  return(scores.reduce(function(sum, score) {
    return sum + score;
  }))/scores.length;
}
